

#### 1.哈希表



```C++
class Solution {
public:
    int missingNumber(vector<int>& nums) {
        unordered_set<intt> s;
        for (int num:nums)
            s.insert(num);
        for (int i=0; i<=nums.size(); i++) {
            if (!s.count(i))
                return i;
        }      
        return -1;
    }
};
```



#### 2.位运算

```c++
class Solution {
public:
    bool containsNearbyDuplicate(vector<int>& nums, int k) {
        unordered_set<intt> s;
        int res = 0;
        for (int num:nums)
            res ^= num;
        for (int i=0; i<nums.size(); i++)
            res ^= i;
        return res;
    }
};
```
