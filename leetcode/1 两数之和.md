 ```
 class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> hashtable;
        for (int i=0; i<nums.size(); i++) {
            int r = target - nums[i];
            if (hashtable.count(r))
                return {hashtable[r], i};
            hashtable[nums[i]] = i;         
        }
        return {};
    }
};
```
